series %>% 
  group_by(series_name) %>%
  filter(series_name == "How I Met Your Mother", season == 1) %>% 
  summarise(correlacao_linear = cor(UserVotes, UserRating, 
                                      method = "pearson"), 
            correlacao_kendall = cor(UserVotes, UserRating, 
                                       method = "kendall"))
